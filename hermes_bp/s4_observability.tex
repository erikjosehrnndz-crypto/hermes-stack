\subsection{Reglas de alerta configuradas}

\begin{lstlisting}[language=yaml,caption=config/alerts.yml — reglas críticas]
groups:
  - name: hermes_stack
    rules:
      - alert: HermesDown
        expr: up{job="hermes"} == 0
        for: 2m
        labels:
          severity: critical
        annotations:
          summary: "Hermes Agent caído"

      - alert: LiteLLMDown
        expr: up{job="litellm"} == 0
        for: 2m
        labels:
          severity: critical

      - alert: HighMemoryUsage
        expr: |
          container_memory_usage_bytes{name=~"hermes.*|litellm.*"}
          / container_spec_memory_limit_bytes > 0.85
        for: 5m
        labels:
          severity: warning

      - alert: HighCPULoad
        expr: 100 - (avg by(instance)(rate(node_cpu_seconds_total
              {mode="idle"}[5m])) * 100) > 85
        for: 10m
        labels:
          severity: warning
\end{lstlisting}

\section{Loop de Auto-Curación}

El sistema implementa tres capas de auto-recuperación:

\begin{table}[h]
\centering
\caption{Capas de auto-curación y tiempos de respuesta}
\begin{tabularx}{\textwidth}{llXr}
\toprule
\textbf{Capa} & \textbf{Herramienta} & \textbf{Acción} & \textbf{Tiempo} \\
\midrule
1 — Healthcheck & Docker Engine    & Marca container unhealthy tras 3 fallos & 90 s \\
2 — Restart     & Autoheal         & Reinicia container unhealthy              & 30--120 s \\
3 — Daemon      & docker-watchdog  & Verifica y reinicia el daemon Docker      & 60 s \\
4 — CI/CD       & GitHub Actions   & Rollback a commit anterior si deploy falla & 3 min \\
\bottomrule
\end{tabularx}
\end{table}

\section{Runbook Operativo}

\subsection{Despliegue manual de emergencia}

\begin{lstlisting}[language=bash,caption=Despliegue manual desde el VPS]
cd /root

# Actualizar codigo
git pull origin main

# Reconstruir y levantar
docker compose up -d --build

# Verificar salud
source .env
curl -f -H "Authorization: Bearer $LITELLM_MASTER_KEY" \
     http://127.0.0.1:4000/health
curl -f http://127.0.0.1:8080/health
\end{lstlisting}

\subsection{Rollback manual}

\begin{lstlisting}[language=bash,caption=Rollback al commit anterior]
cd /root
git log --oneline -5          # Identificar el commit objetivo
git checkout HEAD~1           # O un hash específico
docker compose up -d --build  # Reconstruir con versión anterior

# Verificar y confirmar
curl -f http://127.0.0.1:8080/health && echo "Rollback exitoso"
\end{lstlisting}

\subsection{Rotación de claves}

\textbf{Rotación de \envvar{LITELLM\_MASTER\_KEY}:}

\begin{lstlisting}[language=bash]
# 1. Generar nueva clave
NEW_KEY=$(openssl rand -hex 32)

# 2. Actualizar /root/.env
sed -i "s/^LITELLM_MASTER_KEY=.*/LITELLM_MASTER_KEY=$NEW_KEY/" /root/.env

# 3. Actualizar el secret en GitHub Actions (via gh CLI o UI)
gh secret set LITELLM_MASTER_KEY --body "$NEW_KEY" -R erikjosehrnndz/hermes-stack

# 4. Reiniciar LiteLLM para aplicar nueva clave
docker compose restart litellm
\end{lstlisting}

\subsection{Limpieza de imágenes Docker}

Ejecutar mensualmente para liberar espacio en disco:

\begin{lstlisting}[language=bash,caption=Limpieza de imágenes y volúmenes huérfanos]
# Imágenes sin uso
docker image prune -a --filter "until=720h" -f

# Volúmenes huérfanos (precaución: revisar antes)
docker volume ls -qf dangling=true
docker volume prune -f

# Ver espacio recuperado
docker system df
\end{lstlisting}

\subsection{Verificación rápida de salud del stack}

\begin{lstlisting}[language=bash,caption=Health check completo del stack]
#!/bin/bash
echo "=== Estado de contenedores ==="
docker compose ps

echo "=== Health checks directos ==="
curl -sf http://127.0.0.1:8080/health && echo " Hermes: OK" || echo " Hermes: FAIL"
curl -sf http://127.0.0.1:9000/docs   && echo " Whisper: OK" || echo " Whisper: FAIL"
curl -sf http://127.0.0.1:9090/-/ready && echo " Prometheus: OK" || echo " Prometheus: FAIL"
curl -sf http://127.0.0.1:3000/api/health && echo " Grafana: OK" || echo " Grafana: FAIL"

echo "=== Uso de recursos ==="
docker stats --no-stream --format \
  "table {{.Name}}\t{{.CPUPerc}}\t{{.MemUsage}}"
\end{lstlisting}

\subsection{Alertas y escalación}

\begin{table}[h]
\centering
\caption{Matriz de alertas y acciones}
\begin{tabularx}{\textwidth}{llX}
\toprule
\textbf{Alerta}      & \textbf{Severidad} & \textbf{Acción inmediata} \\
\midrule
HermesDown           & Critical & Verificar logs: \code{docker logs hermes-agent} → reiniciar \\
LiteLLMDown          & Critical & Verificar config \code{litellm.yaml} y API keys en \code{.env} \\
HighMemoryUsage      & Warning  & Revisar \code{docker stats} → considerar límite de memoria \\
HighCPULoad          & Warning  & Identificar proceso con \code{top} → escalar VPS si persiste \\
WhisperDown          & Warning  & Reiniciar: \code{docker compose restart whisper-stt} \\
\bottomrule
\end{tabularx}
\end{table}
